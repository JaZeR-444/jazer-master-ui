Usage Example

<!-- Icon Button Group -->
<div class="icon-btn-group">
    <button class="icon-btn" title="Bold">
        <svg>...</svg>
    </button>
    <button class="icon-btn active" title="Italic">
        <svg>...</svg>
    </button>
    <button class="icon-btn" title="Underline">
        <svg>...</svg>
    </button>
</div>

<!-- Vertical Group -->
<div class="icon-btn-group icon-btn-group--vertical">
    <button class="icon-btn">...</button>
</div>

<!-- Floating Action Button -->
<button class="fab" title="Add">
    <svg>...</svg>
</button>
