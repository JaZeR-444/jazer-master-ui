// State let mouseX = 0, mouseY = 0; let currentBgColor = '#1a0a0a'; let currentCursorColor = '#00ffff'; let isTransitioning = false; // Elements const customCursor = document.getElementById('customCursor'); const bgDetector = document.getElementById('bgDetector'); const bgColorEl = document.getElementById('bgColor'); const cursorColorEl = document.getElementById('cursorColor'); const contrastModeEl = document.getElementById('contrastMode'); const contrastThresholdEl = document.getElementById('contrastThreshold'); const colorTransitionSpeedEl = document.getElementById('colorTransitionSpeed'); const sensitivityEl = document.getElementById('sensitivitySlider'); const primaryColorEl = document.getElementById('primaryColor'); const secondaryColorEl = document.getElementById('secondaryColor'); // Initialize document.addEventListener('DOMContentLoaded', () => { setupEventListeners(); updateStats(); }); // Setup event listeners function setupEventListeners() { // Mouse events document.addEventListener('mousemove', handleMouseMove); document.addEventListener('mouseenter', handleMouseEnter); document.addEventListener('mouseleave', handleMouseLeave); // Control inputs document.getElementById('detectionMode').addEventListener('change', (e) => { config.detectionMode = e.target.value; updateDetectionMode(); }); document.getElementById('contrastThreshold').addEventListener('input', (e) => { config.contrastThreshold = parseInt(e.target.value); }); document.getElementById('colorTransitionSpeed').addEventListener('input', (e) => { config.colorTransitionSpeed = parseInt(e.target.value); }); document.getElementById('primaryColor').addEventListener('input', (e) => { config.primaryColor = e.target.value; }); document.getElementById('secondaryColor').addEventListener('input', (e) => { config.secondaryColor = e.target.value; }); document.getElementById('sensitivitySlider').addEventListener('input', (e) => { config.sensitivity = parseInt(e.target.value); }); // Action buttons document.getElementById('toggleEffect').addEventListener('click', toggleEffect); document.getElementById('resetSettings').addEventListener('click', resetSettings); document.getElementById('randomizeColors').addEventListener('click', randomizeColors); } // Handle mouse movement function handleMouseMove(e) { mouseX = e.clientX; mouseY = e.clientY; // Update custom cursor position customCursor.style.left = mouseX + 'px'; customCursor.style.top = mouseY + 'px'; // Detect background color if (config.effectEnabled) { detectBackgroundColor(); } } // Handle mouse enter function handleMouseEnter(e) { if (config.effectEnabled) { detectBackgroundColor(); } } // Handle mouse leave function handleMouseLeave(e) { if (config.effectEnabled) { // Reset to default when leaving page currentBgColor = '#1a0a0a'; updateCursorColor(); } } // Detect background color function detectBackgroundColor() { const element = document.elementFromPoint(mouseX, mouseY); if (!element) { currentBgColor = '#1a0a0a'; updateCursorColor(); return; } let bgColor = window.getComputedStyle(element).backgroundColor; // Convert to RGB const rgb = bgColor.match(/\d+/g); if (rgb) { const r = parseInt(rgb[1], 16); const g = parseInt(rgb[2], 16); const b = parseInt(rgb[3], 16); currentBgColor = `rgb(${r}, ${g}, ${b})`; } // Calculate luminance const luminance = (0.299 * r + 0.587 * g + 0.114 * b) / 255; // Update based on detection mode switch (config.detectionMode) { case 'luminosity': // Use luminance directly updateCursorColor(); break; case 'contrast': // Invert if below threshold if (luminance < config.contrastThreshold / 100) { currentCursorColor = '#ffffff'; } else { currentCursorColor = '#000000'; } updateCursorColor(); break; case 'inverted': // Always invert currentCursorColor = invertColor(currentBgColor); updateCursorColor(); break; case 'adaptive': // Smart adaptation if (luminance < 0.2) { currentCursorColor = '#ffffff'; } else if (luminance > 0.8) { currentCursorColor = '#000000'; } else { currentCursorColor = '#00ffff'; } updateCursorColor(); break; } updateStats(); } // Update cursor color function updateCursorColor() { if (isTransitioning) return; isTransitioning = true; customCursor.style.background = `radial-gradient(circle, ${currentCursorColor}, ${currentCursorColor})`; customCursor.style.boxShadow = `0 0 20px ${currentCursorColor}`; setTimeout(() => { isTransitioning = false; }, config.colorTransitionSpeed); } // Update stats display function updateStats() { bgColorEl.textContent = currentBgColor; cursorColorEl.textContent = currentCursorColor; const modeLabels = { 'luminosity': 'Luminosity', 'contrast': 'Contrast', 'inverted': 'Inverted', 'adaptive': 'Adaptive' };
