Track Progress Glow Trail

Progress bar with glowing trail effect.

Code

<div class="progress-bar">
    <div class="progress-fill"></div>
</div>

.progress-bar {
    width: 100%;
    height: 8px;
    background: rgba(255, 255, 255, 0.1);
    border-radius: 4px;
    position: relative;
    overflow: hidden;
    margin: 20px 0;
}

.progress-fill {
    height: 100%;
    background: linear-gradient(135deg, #00f2ea 0%, #ff006e 50%, #9333ea 100%);
    border-radius: 4px;
    width: 0%;
    position: relative;
    animation: progress 3s ease-in-out infinite;
}

.progress-fill::after {
    content: '';
    position: absolute;
    top: 0;
    left: 0;
    right: 0;
    bottom: 0;
    background: linear-gradient(90deg, transparent, rgba(255, 255, 255, 0.8), transparent);
    animation: glow 1.5s ease-in-out infinite;
}

@keyframes progress {
    0% { width: 0%; }
    100% { width: 100%; }
}

@keyframes glow {
    0% { transform: translateX(-100%); }
    100% { transform: translateX(100%); }
}
